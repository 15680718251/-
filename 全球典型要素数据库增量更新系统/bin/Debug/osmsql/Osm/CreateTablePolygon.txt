CREATE TABLE polygon
(
objectid NUMBER(30),   
osmid NUMBER(30), 
username VARCHAR2(30),
userid NUMBER(30),  
versionid NUMBER(10),   
starttime VARCHAR2(30),
endtime VARCHAR2(30),
changeset VARCHAR2(300), 
fc VARCHAR2(1000),  
dsg VARCHAR2(1000),  
tags VARCHAR2(3000),   
shape SDO_GEOMETRY NOT NULL, 
visible  NUMBER(10), 
source NUMBER(30),  
trustvalue  FLOAT,
userreputation FLOAT,  
pointsid VARCHAR2(4000),     
issimple NUMBER(10) ,  
Nationelename VARCHAR2(30),
Nationcode VARCHAR2(30),
PRIMARY KEY (objectid),
constraint osmidversionid_unique unique(osmid ,versionid )
)
