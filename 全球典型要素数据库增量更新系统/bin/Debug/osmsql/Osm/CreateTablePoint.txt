CREATE TABLE point(  
objectid NUMBER(30),  
osmid NUMBER(30), 
username VARCHAR2(30),
userid NUMBER(30),  
lat FLOAT NOT NULL,  
lon FLOAT NOT NULL, 
versionid NUMBER(10), 
starttime VARCHAR2(30),
endtime VARCHAR2(30),
changeset NUMBER(30), 
fc VARCHAR2(1000),  
dsg VARCHAR2(100),  
tags VARCHAR2(3000),  
shape SDO_GEOMETRY NOT NULL, 
visible NUMBER(10), 
source NUMBER(30),  
issimple NUMBER(10) ,
Nationelename VARCHAR2(30),
Nationcode VARCHAR2(30),
PRIMARY KEY (objectid),
constraint osmidversionid_unique unique(osmid ,versionid )
)
