CREATE TABLE osmpoint
(
  osmid character varying(20) NOT NULL,
  "user" character varying(50),
  uid character varying(50),
  lat double precision NOT NULL,
  lon double precision NOT NULL,
  "version" smallint,
  changeset character varying(20),
  "timestamp" timestamp without time zone NOT NULL,
  fc character varying(20),
  dsg character varying(20),
  code character(20),
  gbcode character varying(20),
  gbdes character varying(20),
  tags text,
  bz character varying(50),
  "name" character varying(200),
  name_en character varying(200),
  name_zh character varying(200),
  shape geometry,
  mergeid varchar(20) default '0',
  targetLayer varchar(20) default null,
  OReputation float default '0',
  UReputation float default '0',	
  CONSTRAINT nodes_pkey PRIMARY KEY (osmid),
  CONSTRAINT enforce_dims_shape CHECK (st_ndims(shape) = 2),
  CONSTRAINT enforce_geotype_shape CHECK (geometrytype(shape) = 'POINT'::text OR shape IS NULL),
  CONSTRAINT enforce_srid_shape CHECK (st_srid(shape) = 4326)
)
